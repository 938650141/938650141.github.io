给n个挑战关卡，每关至少需要b[i]体力挑战，挑战之前x=min(x,b[i])，挑战需要消耗a[i]体力，无论挑战无论是否完成会加c[i]体力

dp[i][j]表示第i个关卡完成了j个挑战所能剩余的最大体力
转移

dp[i][j]={dp[i−1][j]+c[i](dp[i−1][j]>0)max(dp[i][j],min(dp[i−1][j−1],b[i])−a[i]+c[i])}dp[i][j]=\{ dp[i-1][j]+c[i] (dp[i-1][j]>0) \\ max(dp[i][j],min(dp[i-1][j-1],b[i])-a[i]+c[i]) \} dp[i][j]={dp[i−1][j]+c[i](dp[i−1][j]>0)max(dp[i][j],min(dp[i−1][j−1],b[i])−a[i]+c[i])}

#include<bits/stdc++.h>
using namespace std;

long long dp[1005][1005];
long long a[1005],b[1005],c[1005];

int main(){
	int t;
	cin>>t;
	while(t--){
		int n,m;
		scanf("%d%d",&n,&m);
		for(int i=1;i<=n;i++) scanf("%lld%lld%lld",&a[i],&b[i],&c[i]);
		memset(dp,-1,sizeof(dp));
		dp[0][0]=m;
		int ans=0;
		for(int i=1;i<=n;i++){
			for(int j=0;j<=i;j++){
				if(dp[i-1][j]>0) dp[i][j]=dp[i-1][j]+c[i];
				if(j>0 && dp[i-1][j-1]>a[i] && b[i]>a[i]) dp[i][j]=max(dp[i][j],min(b[i],dp[i-1][j-1])-a[i]+c[i]);
			}
		}
		for(int i=n;i>=0;i--) if(dp[n][i]>0){
			cout<<i<<"\n";
			break;
		}
	}
	return 0;
}

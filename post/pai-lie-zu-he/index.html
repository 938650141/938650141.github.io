n个物品分给m个人，每人最少分1个
C(n-1,m-1)，n-1个空选出m-1个即可

#include<bits/stdc++.h>
using namespace std;

int f[1000005];
int inv[1000005];
const int mod=998244353;

int qpow(int a,int b){
	a%=mod;
	int ans=1;
	while(b){
		if(b&1) ans=((long long)ans*a)%mod;
		b>>=1;
		a=((long long)a*a)%mod;
	}
	return ans;
}

int main(){
	f[0]=1;
	inv[0]=1;
	for(int i=1;i<=1000000;i++) f[i]=((long long)f[i-1]*i)%mod;
	for(int i=1;i<=1000000;i++) inv[i]=qpow(f[i],mod-2);
	int t;
	cin>>t;
	while(t--){
		int n,m;
		scanf("%d%d",&n,&m);
		int ans=1;
		ans=((long long)f[n-1]*inv[m-1])%mod;
		ans=((long long)ans*inv[n-m])%mod;
		printf("%d\n",ans);
	}
	return 0;
}

n个物品分给m个人，允许为空
则人为地填充n个物品，使其满足隔板法要求，再人为地将每个人扣除一个物品，是等价的效果
C(n+m-1,m-1)

#include<bits/stdc++.h>
using namespace std;

int f[2000005];
int inv[2000005];
const int mod=998244353;

int qpow(int a,int b){
	a%=mod;
	int ans=1;
	while(b){
		if(b&1) ans=((long long)ans*a)%mod;
		b>>=1;
		a=((long long)a*a)%mod;
	}
	return ans;
}

int main(){
	f[0]=1;
	inv[0]=1;
	for(int i=1;i<=2000000;i++) f[i]=((long long)f[i-1]*i)%mod;
	for(int i=1;i<=2000000;i++) inv[i]=qpow(f[i],mod-2);
	int t;
	cin>>t;
	while(t--){
		int n,m;
		scanf("%d%d",&n,&m);
		int ans=1;
		ans=((long long)f[n+m-1]*inv[m-1])%mod;
		ans=((long long)ans*inv[n])%mod;
		printf("%d\n",ans);
	}
	return 0;
}

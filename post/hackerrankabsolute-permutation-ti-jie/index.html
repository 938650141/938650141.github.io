[HackerRank]Absolute Permutation题解

2020-09-19 3 min read # HackerRank

给定n和k询问是否存在一个长度为n的不重复数组与从1到n的序列相同位置一一对应元素的绝对值均为k

本题目参考GeeksforGeeks题解*
当k为0是显然只需要输出从1到n的序列即可
更进一步的观察可以相当，n一定是2k的倍数，这样才能满足i和i+k元素的互换，这样i位置是i+k，i+k位置是i，满足了题目的要求。
于是只需要判断好是否满足条件直接模拟一下输出答案即可

#include <bits/stdc++.h>

using namespace std;

vector<string> split_string(string);

// Complete the absolutePermutation function below.
vector<int> absolutePermutation(int n, int k) {
    vector<int>ans;
    if(k==0){
        for(int i=1;i<=n;i++) ans.push_back(i);
        return ans;
    }
    if(n%(k*2)!=0){
        ans.push_back(-1);
        return ans;
    }
    else{
        for(int i=1;i<=n;i++){
            if(((i-1)/k)%2==0) ans.push_back(i+k);
            else ans.push_back(i-k);
        }
        return ans;
    }
    return ans;
}

int main()
{
    ofstream fout(getenv("OUTPUT_PATH"));

    int t;
    cin >> t;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');

    for (int t_itr = 0; t_itr < t; t_itr++) {
        string nk_temp;
        getline(cin, nk_temp);

        vector<string> nk = split_string(nk_temp);

        int n = stoi(nk[0]);

        int k = stoi(nk[1]);

        vector<int> result = absolutePermutation(n, k);

        for (int i = 0; i < result.size(); i++) {
            fout << result[i];

            if (i != result.size() - 1) {
                fout << " ";
            }
        }

        fout << "\n";
    }

    fout.close();

    return 0;
}

vector<string> split_string(string input_string) {
    string::iterator new_end = unique(input_string.begin(), input_string.end(), [] (const char &x, const char &y) {
        return x == y and x == ' ';
    });

    input_string.erase(new_end, input_string.end());

    while (input_string[input_string.length() - 1] == ' ') {
        input_string.pop_back();
    }

    vector<string> splits;
    char delimiter = ' ';

    size_t i = 0;
    size_t pos = input_string.find(delimiter);

    while (pos != string::npos) {
        splits.push_back(input_string.substr(i, pos - i));

        i = pos + 1;
        pos = input_string.find(delimiter, i);
    }

    splits.push_back(input_string.substr(i, min(pos, input_string.length()) - i + 1));

    return splits;
}

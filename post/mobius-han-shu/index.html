#include<cstdio>
using namespace std;
const int N=(int)1e6+5;
int n;
int miu[N];
int v[N];

long long f(int x){
    int ans=0;
    while(x){
        ans+=x%10;
        x/=10;
    }
    return ans;
}

long long g(int x){
    int ans=1;
    while(x){
        ans*=x%10;
        x/=10;
    }
    return ans;
}

int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++) miu[i]=1,v[i]=0;
    for(int i=2;i<=n;i++){
        if(v[i]) continue;
        miu[i]=-1;
        for(int j=2*i;j<=n;j+=i){
            v[j]=1;
            if((j/i)%i==0) miu[j]=0;
            else miu[j]*=-1;
        }
    }
    long long ans=0;
    for(int i=1;i<=n;i++){
        for(int j=i;j<=n;j+=i){
            ans+=(long long)miu[i]*(n/i-j/i+1)*g(j);
            ans+=(long long)miu[i]*(j/i)*f(j);
        }
    }
    printf("%lld",ans);
    return 0;
}

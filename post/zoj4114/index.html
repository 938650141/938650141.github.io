n个开关，k轮，每轮m次按下不同开关
询问从初始状态到终态的方案数

只需要确定初始有k个和终态不同的开关数
之后直接转移即可
dp[i][j]代表第i轮有j个开关与终态不同

#include<bits/stdc++.h>
using namespace std;

const int mod=(int)998244353;

int qpow(int a,int b){
	a%=mod;
	int ans=1;
	while(b){
		if(b&1) ans=((long long)ans*a)%mod;
		a=((long long)a*a)%mod;
		b>>=1;
	}
	return ans;
}

int fec[1005];
int inv[1005];
string s1,s2;
int n,m,k;
long long dp[205][205];

int C(int a,int b){
	return ((long long)fec[a]*inv[b]%mod)*inv[a-b]%mod;
}

int main(){
	fec[0]=1;
	inv[0]=1;
	for(int i=1;i<=1000;i++) fec[i]=((long long)fec[i-1]*i)%mod;
	for(int i=1;i<=1000;i++) inv[i]=qpow(fec[i],mod-2);
	int t;
	cin>>t;
	while(t--){
		cin>>n>>k>>m;
		cin>>s1>>s2;
		memset(dp,0,sizeof(dp));
		int cnt=0;
		for(int i=0;i<n;i++) cnt+=(s1[i]!=s2[i]);
		dp[0][cnt]=1;
		for(int i=0;i<k;i++){
			for(int j=0;j<=n;j++){
				for(int k=0;k<=m;k++){
					if(j>=k && n-j>=m-k)
						dp[i+1][j-k+m-k]+=((long long)(dp[i][j]*C(j,k)%mod)*C(n-j,m-k))%mod,dp[i+1][j-k+m-k]%=mod;
				}
			}
		}
		cout<<dp[k][0]%mod;
		if(t!=0) cout<<"\n";
	}
}
